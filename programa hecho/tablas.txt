import pyodbc
import traceback

SERVER = "10.121.5.80"
DATABASE = "ECU911DMS"
UID = r"ereporte"
PWD = "3st@distic@"


DRIVER_CANDIDATES = [
    "ODBC Driver 18 for SQL Server"
  
]

def try_connect(conn_str):
    try:
        conn = pyodbc.connect(conn_str, timeout=10)
        return conn
    except Exception as e:
        return e

def list_tables(conn):
    try:
        cur = conn.cursor()
        cur.execute("""
            SELECT TABLE_SCHEMA, TABLE_NAME
            FROM INFORMATION_SCHEMA.TABLES
            WHERE TABLE_TYPE='BASE TABLE'
            ORDER BY TABLE_SCHEMA, TABLE_NAME
        """)
        rows = cur.fetchall()
        if not rows:
            print("No se pudo encoontrar tablas.")
            return
        print("Tablas encontradas:")
        for sch, name in rows:
            print(f" - {sch}.{name}")
    except Exception:
        print("Error al listar tablas:")
        traceback.print_exc()

def attempt_odbc_only():
    for driver in DRIVER_CANDIDATES:
        print(f"\nProbando driver: {driver}")

        # ODBC con UID/PWD
        conn_str = (
            f"DRIVER={{{driver}}};"
            f"SERVER={SERVER};"
            f"DATABASE={DATABASE};"
            f"UID={UID};"
            f"PWD={PWD};"
            "Encrypt=yes;"
            "TrustServerCertificate=yes;"
        )

        print(" -> Intentando conexion ODBC con usuario/clave...")
        r = try_connect(conn_str)
        if isinstance(r, pyodbc.Connection):
            print("Conectado OK con:", driver)
            list_tables(r)
            r.close()
            return True
        else:
            print("   Fallo:", type(r).__name__, str(r))

    print("\nNo se pudo conectar con ningun driver ODBC probado.")
    return False

if __name__ == "__main__":
    print("Iniciando prueba de conexion a SQL Server (solo ODBC con UID/PWD)")
    ok = attempt_odbc_only()
    if not ok:
        print("\n-- No valio :(")
    else:
        print("\nConexion OK — ya listaste las tablas.")
